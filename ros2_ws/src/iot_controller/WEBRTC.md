export THING_NAME=rover1
export AWS_ACCESS_KEY_ID=

export AWS_WEBRTC_CHANNEL=arn:aws:kinesisvideo:ap-south-1:783058511096:channel/iot-kinesis-demo/1731498094389

ros2 run iot_controller mqtt_listener
